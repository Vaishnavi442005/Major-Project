export type AuthUser = {
  email: string;
  name?: string;
  loggedIn: boolean;
  mode: "email" | "google" | "guest";
};

export function setAuth(user: AuthUser) {
  localStorage.setItem("aura_auth", JSON.stringify(user));
}

export function clearAuth() {
  localStorage.removeItem("aura_auth");
  sessionStorage.removeItem("aura_guest");
}

export function setGuestMode() {
  sessionStorage.setItem("aura_guest", "1");
}

export function isGuest(): boolean {
  return sessionStorage.getItem("aura_guest") === "1";
}

export function getAuth(): AuthUser | null {
  if (isGuest()) {
    return { email: "guest", name: "Guest", loggedIn: true, mode: "guest" };
  }
  const raw = localStorage.getItem("aura_auth");
  if (!raw) return null;
  try { return JSON.parse(raw); } catch { return null; }
}

export default function ChatPage() {
  const [match, params] = useRoute("/chat/:id");
  const [, setLocation] = useLocation();
  const queryClient = useQueryClient();
  const guest = isGuest();
  const conversationId = !guest && match && params?.id ? parseInt(params.id) : null;

  const [input, setInput] = useState("");
  const [sidebarOpen, setSidebarOpen] = useState(false);
  const [selectedImage, setSelectedImage] = useState<{ base64: string; mimeType: string; previewUrl: string } | null>(null);
  const [guestMessages, setGuestMessages] = useState<StreamMessage[]>([]);
  const messagesEndRef = useRef<HTMLDivElement>(null);
  const fileInputRef = useRef<HTMLInputElement>(null);

  const { data: conversation, isLoading: isLoadingConv } = useGetOpenaiConversation(
    conversationId as number, { query: { enabled: !!conversationId && !guest } }
  );
  const createMutation = useCreateOpenaiConversation();
  const { startStream, startGuestStream, optimisticMessages, isStreaming } = useChatStream();

  const dbMessages = (conversation?.messages || []).map(m => ({
    id: m.id.toString(), role: m.role as "user" | "assistant",
    content: m.content, imageUrl: (m as any).imageUrl ?? undefined,
  }));

  const displayMessages = guest ? guestMessages : [...dbMessages, ...optimisticMessages];

  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [displayMessages]);

  const handleImageSelect = (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file) return;
    if (!file.type.startsWith("image/")) { toast.error("Please select an image file"); return; }
    const reader = new FileReader();
    reader.onload = () => {
      const dataUrl = reader.result as string;
      setSelectedImage({ base64: dataUrl.split(",")[1], mimeType: file.type, previewUrl: dataUrl });
    };
    reader.readAsDataURL(file);
    e.target.value = "";
  };

  const handleSubmit = async (e?: React.FormEvent, suggestionText?: string) => {
    e?.preventDefault();
    const textToSend = suggestionText || input;
    if ((!textToSend.trim() && !selectedImage) || isStreaming) return;
    const imageToSend = selectedImage;
    setInput(""); setSelectedImage(null);

    if (guest) {
      const currentMessages = guestMessages.filter(m => !m.isStreaming);
      await startGuestStream(currentMessages, textToSend || "Please analyze this image.",
        imageToSend?.base64, imageToSend?.mimeType, imageToSend?.previewUrl,
        (msgs) => setGuestMessages(msgs));
      return;
    }

    try {
      let activeId = conversationId;
      if (!activeId) {
        const title = textToSend.slice(0, 40);
        const newConv = await createMutation.mutateAsync({ data: { title } });
        activeId = newConv.id;
        queryClient.invalidateQueries({ queryKey: getListOpenaiConversationsQueryKey() });
        setLocation(`/chat/${activeId}`, { replace: true });
      }
      await startStream(activeId, textToSend, imageToSend?.base64, imageToSend?.mimeType, imageToSend?.previewUrl);
    } catch { toast.error("Failed to process your request"); }
  };

  const canSend = (input.trim() || selectedImage) && !isStreaming;

  return (
    <div className="flex h-screen bg-background overflow-hidden">
      <div className={`fixed inset-y-0 left-0 z-50 w-72 md:relative md:translate-x-0 transform transition-transform ${sidebarOpen ? "translate-x-0" : "-translate-x-full"}`}>
        <Sidebar onCloseMobile={() => setSidebarOpen(false)} />
      </div>
      <div className="flex-1 flex flex-col min-w-0 h-full">
        <header className="h-14 flex items-center justify-between px-4 border-b bg-background/80">
          <div className="flex items-center">
            <Button variant="ghost" size="icon" className="md:hidden" onClick={() => setSidebarOpen(true)}>
              <Menu className="w-5 h-5" />
            </Button>
            <h2 className="font-semibold">{guest ? "Guest Session" : (conversation?.title || "New Assessment")}</h2>
          </div>
          <div className="flex items-center text-xs text-muted-foreground bg-muted/50 px-3 py-1.5 rounded-full">
            <Shield className="w-3.5 h-3.5 mr-1.5 text-primary" />
            {guest ? "Guest — chats not saved" : "Secure & Private"}
          </div>
        </header>

        <div className="flex-1 overflow-y-auto p-4 sm:p-6">
          <div className="max-w-3xl mx-auto">
            {displayMessages.length === 0 ? (
              <motion.div initial={{ opacity: 0, y: 20 }} animate={{ opacity: 1, y: 0 }}
                className="flex flex-col items-center justify-center min-h-[60vh] text-center px-4">
                <Sparkles className="w-8 h-8 text-primary mb-6" />
                <h2 className="text-2xl font-bold mb-2">How can I help you today?</h2>
                <p className="text-muted-foreground mb-8">
                  {guest ? "Guest mode — your chat won't be saved." : "Describe your symptoms or ask about medications."}
                </p>
                <div className="grid grid-cols-1 sm:grid-cols-3 gap-3 w-full max-w-2xl">
                  {SUGGESTIONS.map((item, i) => (
                    <button key={i} onClick={() => handleSubmit(undefined, item.text)}

                      <item.icon className="w-6 h-6 mb-3 text-muted-foreground" />
                      <span className="font-medium">{item.text}</span>
                    </button>
                  ))}
                </div>
              </motion.div>
            ) : (

              </div>

        </div>

        <div className="p-4 shrink-0">
          <div className="max-w-3xl mx-auto">
            {selectedImage && (

import { useState, useRef, useCallback } from "react";
import { useQueryClient } from "@tanstack/react-query";
import { getGetOpenaiConversationQueryKey } from "@workspace/api-client-react";

export type StreamMessage = {
  id: string;
  role: "user" | "assistant";
  content: string;
  imageUrl?: string;
  isStreaming?: boolean;
};

export function useChatStream() {
  const [optimisticMessages, setOptimisticMessages] = useState<StreamMessage[]>([]);
  const [isStreaming, setIsStreaming] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const abortControllerRef = useRef<AbortController | null>(null);
  const queryClient = useQueryClient();

  // Authenticated users: streams and saves to database
  const startStream = useCallback(async (
    conversationId: number, content: string,
    imageBase64?: string, imageMimeType?: string, imagePreviewUrl?: string,
  ) => {
    setIsStreaming(true);
    const userMsgId = `user-${Date.now()}`;
    const assistantMsgId = `assistant-${Date.now()}`;
    setOptimisticMessages([
      { id: userMsgId, role: "user", content, imageUrl: imagePreviewUrl },
      { id: assistantMsgId, role: "assistant", content: "", isStreaming: true }
    ]);
    abortControllerRef.current = new AbortController();
    try {
      const body: Record<string, string> = { content };
      if (imageBase64) body.imageBase64 = imageBase64;
      if (imageMimeType) body.imageMimeType = imageMimeType;
      const response = await fetch(`/api/openai/conversations/${conversationId}/messages`, {
        method: "POST", headers: { "Content-Type": "application/json" },
        body: JSON.stringify(body), signal: abortControllerRef.current.signal,
      });
      const reader = response.body!.getReader();
      const decoder = new TextDecoder();
      let done = false, fullContent = "";
      while (!done) {
        const { value, done: readerDone } = await reader.read();
        done = readerDone;
        if (value) {
          for (const line of decoder.decode(value, { stream: true }).split("\n")) {
            if (line.startsWith("data: ")) {
              const data = JSON.parse(line.slice(6));
              if (data.done) done = true;
              else if (data.content) {
                fullContent += data.content;
                setOptimisticMessages(prev => prev.map(msg =>
                  msg.id === assistantMsgId ? { ...msg, content: fullContent } : msg));
              }
            }
          }
        }
      }
      await queryClient.invalidateQueries({ queryKey: getGetOpenaiConversationQueryKey(conversationId) });
    } catch (err: any) {
      if (err.name !== "AbortError") setError(err.message);
    } finally {
      setIsStreaming(false);
      setTimeout(() => setOptimisticMessages([]), 500);
    }
  }, [queryClient]);

  // Guest users: streams without saving to database
  const startGuestStream = useCallback(async (
    messages: StreamMessage[], content: string,
    imageBase64?: string, imageMimeType?: string, imagePreviewUrl?: string,
    onNewMessage?: (msgs: StreamMessage[]) => void,
  ) => {
    setIsStreaming(true);
    abortControllerRef.current = new AbortController();
    const userMsg: StreamMessage = { id: `user-${Date.now()}`, role: "user", content, imageUrl: imagePreviewUrl };
    const assistantMsg: StreamMessage = { id: `assistant-${Date.now()}`, role: "assistant", content: "", isStreaming: true };
    const updatedMessages = [...messages, userMsg, assistantMsg];
    onNewMessage?.(updatedMessages);
    try {
      const history = messages.map(m => ({ role: m.role, content: m.content, imageUrl: m.imageUrl }));
      history.push({ role: "user", content, imageUrl: imagePreviewUrl });
      const body: Record<string, any> = { messages: history };
      if (imageBase64) body.imageBase64 = imageBase64;
      if (imageMimeType) body.imageMimeType = imageMimeType;
      const response = await fetch(`/api/openai/guest/chat`, {
        method: "POST", headers: { "Content-Type": "application/json" },
        body: JSON.stringify(body), signal: abortControllerRef.current.signal,
      });
      const reader = response.body!.getReader();
      const decoder = new TextDecoder();
      let done = false, fullContent = "";
      while (!done) {
        const { value, done: readerDone } = await reader.read();
        done = readerDone;
        if (value) {
          for (const line of decoder.decode(value, { stream: true }).split("\n")) {
            if (line.startsWith("data: ")) {
              const data = JSON.parse(line.slice(6));
              if (data.done) done = true;
              else if (data.content) {
                fullContent += data.content;
                onNewMessage?.(updatedMessages.map(m =>
                  m.id === assistantMsg.id ? { ...m, content: fullContent } : m));
              }
            }
          }
        }
      }
      onNewMessage?.(updatedMessages.map(m =>
        m.id === assistantMsg.id ? { ...m, content: fullContent, isStreaming: false } : m));
    } catch (err: any) {
      if (err.name !== "AbortError") setError(err.message);
    } finally { setIsStreaming(false); }
  }, []);

  return { optimisticMessages, isStreaming, error, startStream, startGuestStream,
    stopStream: () => { abortControllerRef.current?.abort(); setIsStreaming(false); } };
}
import { Activity, User } from "lucide-react";

interface MessageBubbleProps {
  role: "user" | "assistant";
  content: string;
  imageUrl?: string;
  isStreaming?: boolean;
}

export function MessageBubble({ role, content, imageUrl, isStreaming }: MessageBubbleProps) {
  const isAssistant = role === "assistant";
  return (
    <motion.div initial={{ opacity: 0, y: 10 }} animate={{ opacity: 1, y: 0 }}
      className={`flex w-full ${isAssistant ? "justify-start" : "justify-end"} mb-6`}>
      <div className={`flex max-w-[85%] md:max-w-[75%] ${isAssistant ? "flex-row" : "flex-row-reverse"}`}>
        <div className={`flex-shrink-0 ${isAssistant ? "mr-4" : "ml-4"}`}>
          <div className={`w-8 h-8 rounded-full flex items-center justify-center ${isAssistant ? "bg-gradient-to-br from-primary to-accent text-white" : "bg-secondary"}`}>
            {isAssistant ? <Activity className="w-4 h-4" /> : <User className="w-4 h-4" />}
          </div>
        </div>
        <div className="flex flex-col gap-2">
          {imageUrl && (
            <div className="rounded-2xl overflow-hidden border border-border">
              <img src={imageUrl} alt="Uploaded" className="max-w-xs max-h-64 object-cover" />
            </div>
          )}
          {(content || isStreaming) && (
            <div className={`px-5 py-4 rounded-2xl text-[15px] leading-relaxed ${isAssistant ? "bg-card border text-card-foreground rounded-tl-none" : "bg-primary text-primary-foreground rounded-tr-none"}`}>
              {content ? (
                <div className={`prose prose-sm max-w-none ${isAssistant ? "prose-slate" : "prose-invert"}`}>
                  <ReactMarkdown remarkPlugins={[remarkGfm]}>{content}</ReactMarkdown>
                </div>
              ) : isStreaming ? (
                <div className="flex space-x-1 items-center h-5">
                  <div className="w-1.5 h-1.5 bg-primary/40 rounded-full animate-bounce [animation-delay:-0.3s]"></div>
                  <div className="w-1.5 h-1.5 bg-primary/60 rounded-full animate-bounce [animation-delay:-0.15s]"></div>
                  <div className="w-1.5 h-1.5 bg-primary/80 rounded-full animate-bounce"></div>
                </div>
              ) : null}
            </div>
          )}
        </div>
      </div>
    </motion.div>
  );
}

// POST send message (authenticated — saves to DB, streams AI response)
router.post("/conversations/:id/messages", async (req, res) => {
  const id = parseInt(req.params.id);
  const { content, imageBase64, imageMimeType } = req.body;
  const imageUrl = imageBase64 && imageMimeType ? `data:${imageMimeType};base64,${imageBase64}` : null;

  await db.insert(messages).values({ conversationId: id, role: "user", content, imageUrl: imageUrl ?? undefined });
  const history = await db.select().from(messages).where(eq(messages.conversationId, id)).orderBy(messages.createdAt);

  const chatMessages: any[] = [{ role: "system", content: SYSTEM_PROMPT }];
  for (const m of history) {
    if (m.role === "user" && m.imageUrl)
      chatMessages.push({ role: "user", content: [{ type: "text", text: m.content }, { type: "image_url", image_url: { url: m.imageUrl } }] });
    else
      chatMessages.push({ role: m.role, content: m.content });
  }

  res.setHeader("Content-Type", "text/event-stream");
  res.setHeader("Cache-Control", "no-cache");

  let fullResponse = "";
  const stream = await openai.chat.completions.create({ model: "gpt-5.2", messages: chatMessages, stream: true });
  for await (const chunk of stream) {
    const content = chunk.choices[0]?.delta?.content;
    if (content) { fullResponse += content; res.write(`data: ${JSON.stringify({ content })}\n\n`); }
  }
  await db.insert(messages).values({ conversationId: id, role: "assistant", content: fullResponse });
  res.write(`data: ${JSON.stringify({ done: true })}\n\n`);
  res.end();
});

import { integer, pgTable, serial, text, timestamp } from "drizzle-orm/pg-core";
import { conversations } from "./conversations";

// Conversations table
export const conversations = pgTable("conversations", {
  id: serial("id").primaryKey(),
  title: text("title").notNull(),
  createdAt: timestamp("created_at", { withTimezone: true }).defaultNow().notNull(),
});

// Messages table
export const messages = pgTable("messages", {
  id: serial("id").primaryKey(),
  conversationId: integer("conversation_id")
    .notNull()
    .references(() => conversations.id, { onDelete: "cascade" }),
  role: text("role").notNull(),          // "user" or "assistant"
  content: text("content").notNull(),
  imageUrl: text("image_url"),           // base64 data URL for image uploads
  createdAt: timestamp("created_at", { withTimezone: true }).defaultNow().notNull(),
});
